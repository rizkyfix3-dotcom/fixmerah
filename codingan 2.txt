# ==================== VALIDASI NOMOR TELEPON ====================
def validate_phone(phone_input):
    """Validasi dan format nomor telepon"""
    if not phone_input:
        return False, "❌ Masukkan nomor telepon!"
    
    # Hapus spasi dan karakter khusus
    phone = re.sub(r'[^\d\+]', '', phone_input)
    
    # Cek panjang minimal
    if len(phone) < 8:
        return False, "❌ Nomor telepon terlalu pendek!"
    
    # Format yang diterima:
    # +62xxxxxxxxxx, 62xxxxxxxxxx, 08xxxxxxxx, 628xxxxxxxx
    
    # Jika diawali dengan 0, ganti dengan 62
    if phone.startswith('0'):
        phone = '62' + phone[1:]
    elif phone.startswith('8') and len(phone) <= 11:
        phone = '62' + phone
    elif phone.startswith('+'):
        phone = phone[1:]  # Hapus +
    
    # Pastikan hanya angka
    if not phone.isdigit():
        return False, "❌ Nomor telepon harus angka!"
    
    # Tambahkan + di depan
    formatted = '+' + phone
    
    return True, formatted

# ==================== FIXMERAH COMMAND ====================
async def fixmerah_command(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Handle /fixmerah command"""
    user_id = update.effective_user.id
    
    # Cek verifikasi
    if not premium.is_user_verified(user_id):
        await update.message.reply_text(
            f"""❌ *BELUM TERVERIFIKASI!*

Kamu harus verifikasi dulu:
1. Join Channel: {VERIFICATION_CHANNEL_LINK}
2. Join Group: {VERIFICATION_GROUP_LINK}
3. Klik /start untuk verifikasi"""
        )
        return
    
    # Cek maintenance
    if MAINTENANCE and not premium.is_any_owner(user_id):
        await update.message.reply_text("🔧 Bot sedang maintenance. Coba lagi nanti!")
        return
    
    # Cek argumen
    if not context.args:
        await update.message.reply_text(
            "❌ Format: `/fixmerah [nomor_telepon]`\n"
            "Contoh: `/fixmerah +628123456789`",
            parse_mode='Markdown'
        )
        return
    
    phone_input = " ".join(context.args)
    
    # Validasi nomor
    success, result = validate_phone(phone_input)
    if not success:
        await update.message.reply_text(result)
        return
    
    phone_number = result
    
    # Tampilkan proses
    processing_msg = await update.message.reply_text(
        f"""⏳ *MENGIRIM APPEAL...*

📱 Nomor: `{phone_number}`
👤 Pengirim: {update.effective_user.first_name}
🆔 ID: `{user_id}`
⏰ Waktu: {datetime.now().strftime('%H:%M:%S')}

Mohon tunggu..."""
    )
    
    # Kirim appeal via email
    success, message = email.send_appeal(phone_number)
    
    if success:
        # Update counter
        premium.update_fixmerah_count(user_id)
        
        await processing_msg.edit_text(
            f"""✅ *APPEAL TERKIRIM!*

📱 Nomor: `{phone_number}`
✅ Status: Berhasil dikirim ke WhatsApp Support
📧 Email: {email.config.get('email', 'Not set')}
🕐 Waktu: {datetime.now().strftime('%H:%M:%S')}

📨 Appeal telah dikirim ke:
support@support.whatsapp.com

💡 *Tips:* Tunggu 24-48 jam untuk respons dari WhatsApp."""
        )
    else:
        await processing_msg.edit_text(
            f"""❌ *GAGAL MENGIRIM!*

📱 Nomor: `{phone_number}`
❌ Error: {message}

🔧 *SOLUSI:*
1. Periksa koneksi internet
2. Owner: Cek email config (/setemail)
3. Hubungi @Fixmerahbydho jika masalah berlanjut"""
        )

# ==================== FIXMERAHALL COMMAND (BUYER ONLY) ====================
async def fixmerahall_command(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Handle /fixmerahall command (batch sending)"""
    user_id = update.effective_user.id
    
    # Cek akses buyer
    if not premium.is_any_owner(user_id) and str(user_id) not in premium.data["buyer_users"]:
        await update.message.reply_text(
            "❌ *FITUR BUYER ONLY!*\n\n"
            "Fitur ini hanya untuk user buyer.\n"
            "Hubungi @Fixmerahbydho untuk upgrade!"
        )
        return
    
    # Cek verifikasi
    if not premium.is_user_verified(user_id):
        await update.message.reply_text("❌ Belum terverifikasi! Gunakan /start dulu.")
        return
    
    # Cek argumen
    if not context.args:
        await update.message.reply_text(
            "❌ Format: `/fixmerahall [nomor1] [nomor2] ...`\n"
            "Contoh: `/fixmerahall +628123456789 +628987654321`\n\n"
            f"*Maksimal {MAX_BATCH_SIZE} nomor per batch*",
            parse_mode='Markdown'
        )
        return
    
    # Batasi jumlah nomor
    if len(context.args) > MAX_BATCH_SIZE:
        await update.message.reply_text(
            f"❌ Maksimal {MAX_BATCH_SIZE} nomor per batch!"
        )
        return
    
    # Proses batch
    phones = context.args
    valid_phones = []
    invalid_phones = []
    
    # Validasi semua nomor
    for phone in phones:
        success, result = validate_phone(phone)
        if success:
            valid_phones.append(result)
        else:
            invalid_phones.append(f"{phone}: {result}")
    
    if not valid_phones:
        await update.message.reply_text(
            "❌ Tidak ada nomor yang valid!\n" + "\n".join(invalid_phones)
        )
        return
    
    # Tampilkan progress
    progress_msg = await update.message.reply_text(
        f"""⏳ *MEMPROSES BATCH...*

📱 Total: {len(phones)} nomor
✅ Valid: {len(valid_phones)} nomor
❌ Invalid: {len(invalid_phones)} nomor

⏰ Estimasi: {len(valid_phones) * BATCH_DELAY_SECONDS} detik

Memulai pengiriman..."""
    )
    
    # Kirim satu per satu dengan delay
    success_count = 0
    fail_count = 0
    results = []
    
    for i, phone in enumerate(valid_phones, 1):
        try:
            # Update progress
            await progress_msg.edit_text(
                f"""⏳ *MENGIRIM BATCH...*

📊 Progress: {i}/{len(valid_phones)}
📱 Sedang mengirim: `{phone}`
✅ Berhasil: {success_count}
❌ Gagal: {fail_count}

⏰ Delay: {BATCH_DELAY_SECONDS} detik"""
            )
            
            # Kirim appeal
            success, message = email.send_appeal(phone)
            
            if success:
                success_count += 1
                results.append(f"✅ {phone}: Berhasil")
                # Update counter untuk user
                premium.update_fixmerah_count(user_id)
            else:
                fail_count += 1
                results.append(f"❌ {phone}: Gagal - {message}")
            
            # Delay antara pengiriman
            if i < len(valid_phones):
                await asyncio.sleep(BATCH_DELAY_SECONDS)
                
        except Exception as e:
            fail_count += 1
            results.append(f"❌ {phone}: Error - {str(e)}")
    
    # Kirim hasil akhir
    result_text = f"""📊 *BATCH COMPLETE!*

📱 Total diproses: {len(valid_phones)} nomor
✅ Berhasil: {success_count}
❌ Gagal: {fail_count}

⏰ Selesai: {datetime.now().strftime('%H:%M:%S')}

📋 *HASIL DETAIL:*
""" + "\n".join(results[:10])  # Batasi output

    if len(results) > 10:
        result_text += f"\n\n...dan {len(results)-10} hasil lainnya"

    await progress_msg.edit_text(result_text)
